        self.p_lon_min = p_lon_min

        self.p_lon_max = p_lon_max
        self.p_lat_max = p_lat_max

        super().__init__([(p_lon_min, p_lat_min), (p_lon_max, p_lat_max)])

    def __repr__(self):
